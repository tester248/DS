# Boxplot / Violin plot: fare distribution by class
plt.figure(figsize=(8,5))
sns.boxplot(x='class', y='fare', data=dataset, palette='Set2')
plt.title('Fare distribution by Class (boxplot)')
plt.tight_layout()
plt.show()

plt.figure(figsize=(8,5))
sns.violinplot(x='class', y='fare', data=dataset, palette='Set3')
plt.title('Fare distribution by Class (violinplot)')
plt.tight_layout()
plt.show()

# Categorical plots: counts by sex and class
fig, axes = plt.subplots(1, 2, figsize=(12,4))
sns.countplot(x='sex', data=dataset, ax=axes[0], palette='pastel')
axes[0].set_title('Count by Sex')

sns.countplot(x='class', data=dataset, ax=axes[1], palette='muted')
axes[1].set_title('Count by Class')

plt.tight_layout()
plt.show()

# Bar plot: average fare by class
plt.figure(figsize=(6,4))
sns.barplot(x='class', y='fare', data=dataset, estimator=lambda x: sum(x)/len(x))
plt.title('Average Fare by Class')
plt.ylabel('Average Fare')
plt.tight_layout()
plt.show()

# Distribution plot for 'age' (with KDE)
plt.figure(figsize=(10,4))
sns.histplot(dataset['age'].dropna(), kde=True, bins=30, color='C1')
plt.title('Distribution of Passenger Age')
plt.xlabel('Age')
plt.ylabel('Count')
plt.tight_layout()
plt.show()

# Rug plot for age (optional)
plt.figure(figsize=(10,1))
sns.rugplot(dataset['age'].dropna(), height=0.5)
plt.yticks([])
plt.xlabel('Age')
plt.title('Rug plot for Age')
plt.tight_layout()
plt.show()
